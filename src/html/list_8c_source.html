312 bool list_add_all_at(List *list1, List *list2, size_t index)
313 {
314  if (list2->size == 0)
315  return false;
316 
317  if (index > list1->size)
318  return false;
319 
320  /* Link the new nodes together outside of the list so
321  that if anything goes wrong we have don't leave
322  garbage in the actual list. */
323  Node *head = NULL;
324  Node *tail = NULL;
325 
326  if (!link_all_externally(list2, &head, &tail))
327  return false;
328 
329  /* Now we can safely attach the new nodes. */
330  Node *end = get_node_at(list1, index);
331  Node *base = end ? end->prev : get_node_at(list1, index - 1); // calling get node twice is not very efficient
332 
333  if (!end) {
334  list1->tail->next = head;
335  head->prev = list1->tail;
336  list1->tail = tail;
337  } else if (!base) {
338  list1->head->prev = tail;
339  tail->next = list1->head;
340  list1->head = head;
341  } else {
342  head->prev = base;
343  base->next = head;
344  tail->next = end;
345  end->prev = tail;
346  }
347 
348  list1->size += list2->size;
349 
350  return true;
351 }

364 static bool link_all_externally(List *list, Node **h, Node **t)
365 {
366  Node *insert = list->head;
367 
368  size_t i;
369  for (i = 0; i < list->size; i++) {
370  Node *new = list->mem_calloc(1, sizeof(Node));
371 
372  if (!new) {

1005 static INLINE void merge(Node **left, Node **right, size_t l_size,
1006  size_t r_size, int (*cmp) (void const*, void const*))
1007 {
1008  size_t size = r_size + l_size;
1009  size_t l = 0; /* Number of processed elements from the left partition */
1010  size_t r = 0; /* NUmber of processed elements from the right partition. */
1011 
1012  Node *l_part = *left;
1013  Node *r_part = *right;
1014 
1015  size_t i;
1016  for (i = 0; i < size; i++) {
1017  int c = cmp(&(l_part->data), &(r_part->data));
1018 
1019  if ((c == -1 || c == 0)) {
1020  /* The two partitions are already sorted. */
1021  if (i == 0 && size == 2) {
1022  break;
1023  }
1024  /* If all the elements on the left side are in place,
1025  all the elements on the right must be as well. */
1026  if (l == l_size) {
1027  /* Quickly advance the right pointer to the end for
1028  * the remainder of r and exit.*/
1029  for (;r < r_size - 1; r++)
1030  r_part = r_part->next;
1031  *right = r_part;
1032  break;
1033  }
1034  l_part = l_part->next;
1035  l++;
1036  } else {
1037  Node *tmp = (r_part)->next;
1038  link_behind(l_part, r_part);
1039  /* The elements are swapped. Now set head and tail and
1040  * exit the loop. */
1041  if (i == 0 && size == 2) {
1042  *right = l_part;
1043  *left = r_part;
1044  break;
1045  }
1046  r++;
1047  /* All right partition elements are now behind the left
1048  * partition elements.*/
1049  if (r == r_size) {
1050  /* Quickly advance the left partition pointer and set
1051  * the rightmost element as the new head. */
1052  for (;l < l_size - 1; l++)
1053  l_part = l_part->next;
1054  *right = l_part;
1055  break;
1056  }
1057  if (i == 0) {
1058  *left = r_part;
1059  }
1060  r_part = tmp;
1061  }
1062  }
1063 }

1349 static void link_behind(Node *const base, Node *ins)
1350 {
1351  /* link the gap */
1352  if (ins->next != NULL)
1353  ins->next->prev = ins->prev;
1354 
1355  if (ins->prev != NULL)
1356  ins->prev->next = ins->next;
1357 
1358  /* link behind */
1359  if (base->prev == NULL) {
1360  ins->prev = NULL;
1361  ins->next = base;
1362  base->prev = ins;
