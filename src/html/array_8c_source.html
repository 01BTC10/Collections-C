65 Array *array_new_conf(ArrayConf *conf)
66 {
67  Array *ar = conf->mem_calloc(1, sizeof(Array));
68 
69  if (ar == NULL)
70  return NULL;
71 
72  if (conf->exp_factor <= 1)
73  ar->exp_factor = DEFAULT_EXPANSION_FACTOR;
74 
75  /* Needed to avoid an integer overflow on the first resize and
76  * to easily check for any future oveflows. */
77  else if (conf->exp_factor >= MAX_ELEMENTS / conf->capacity)
78  return NULL;
79  else
80  ar->exp_factor = conf->exp_factor;
81 
82  ar->capacity = conf->capacity;
83  ar->mem_alloc = conf->mem_alloc;
84  ar->mem_calloc = conf->mem_calloc;
85  ar->mem_free = conf->mem_free;
86  ar->buffer = ar->mem_calloc(ar->capacity, sizeof(void*));
87 
88  return ar;
89 }

599 static bool expand_capacity(Array *ar)
600 {
601  if (ar->capacity == MAX_ELEMENTS)
602  return false;
603 
604  size_t new_capacity = ar->capacity * ar->exp_factor;
605 
606  /* As long as the capacity is greater that the expansion factor
607  * at the point of overflow, this is check is valid. */
608  if (new_capacity <= ar->capacity)
609  ar->capacity = MAX_ELEMENTS;
610  else
611  ar->capacity = new_capacity;
612 
613  void **new_buff = ar->mem_alloc(new_capacity * sizeof(void*));
614 
615  if (new_buff == NULL)
616  return false;
617 
618  memcpy(new_buff, ar->buffer, ar->size * sizeof(void*));
619 
620  ar->mem_free(ar->buffer);
621  ar->buffer = new_buff;
622 
623  return true;
624 }
