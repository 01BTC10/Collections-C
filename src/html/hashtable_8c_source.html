152 bool hashtable_add(HashTable *table, void *key, void *val)
153 {
154  if (table->size >= table->threshold)
155  resize(table, table->capacity << 1);
156 
157  if (!key)
158  return add_null_key(table, val);
159 
160  const size_t hash = table->hash(key, table->key_len, table->hash_seed);
161  const size_t i = hash & (table->capacity - 1);
162 
163  TableEntry *replace = table->buckets[i];
164 
165  while (replace) {
166  if (table->key_cmp(replace->key, key)) {
167  replace->value = val;
168  return true;
169  }
170  replace = replace->next;
171  }
172 
173  TableEntry *new_entry = table->mem_alloc(sizeof(TableEntry));
174 
175  if (!new_entry)
176  return false;
177 
178  new_entry->key = key;
179  new_entry->value = val;
180  new_entry->hash = hash;
181  new_entry->next = table->buckets[i];
182 
183  table->buckets[i] = new_entry;
184  table->size++;
185 
186  return true;
187 }

390 static bool resize(HashTable *t, size_t new_capacity)
391 {
392  if (t->capacity == MAX_POW_TWO)
393  return false;
394 
395  TableEntry **new_buckets = t->mem_calloc(new_capacity, sizeof(TableEntry));
396  TableEntry **old_buckets = t->buckets;
397 
398  if (!new_buckets)
399  return false;
400 
401  move_entries(old_buckets, new_buckets, t->capacity, new_capacity);
402 
403  t->buckets = new_buckets;
404  t->capacity = new_capacity;
405  t->threshold = t->load_factor * new_capacity;
406 
407  t->mem_free(old_buckets);
408 
409  return true;
410 }

419 static INLINE size_t round_pow_two(size_t n)
420 {
421  if (n >= MAX_POW_TWO)
422  return MAX_POW_TWO;
423 
424  if (n == 0)
425  return 2;
431  n--;
432  n |= n >> 1;
433  n |= n >> 2;
434  n |= n >> 4;
435  n |= n >> 8;
436  n |= n >> 16;
437  n++;
438 
439  return n;
440 }

902 uint64_t hashtable_hash(const void *key, int len, uint32_t seed)
903 {
904  const uint8_t *data = (const uint8_t*) key;
905  const int nblocks = len / 16;
906 
907  uint64_t h1 = seed;
908  uint64_t h2 = seed;
909 
910  const uint64_t c1 = BIG_CONSTANT(0x87c37b91114253d5);
911  const uint64_t c2 = BIG_CONSTANT(0x4cf5ad432745937f);
912 
913  const uint64_t *blocks = (const uint64_t*)(data);
914 
915  int i;
916  for(i = 0; i < nblocks; i++) {
917  uint64_t k1 = blocks[i*2+0];
918  uint64_t k2 = blocks[i*2+1];
919 
920  k1 *= c1;
921  k1 = ROTL64(k1,31);
922  k1 *= c2;
923  h1 ^= k1;
924  h1 = ROTL64(h1,27);
925  h1 += h2;
926  h1 = h1 * 5 + 0x52dce729;
927 
928  k2 *= c2;
929  k2 = ROTL64(k2,33);
930  k2 *= c1;
931  h2 ^= k2;
932  h2 = ROTL64(h2,31);
933  h2 += h1;
934  h2 = h2 * 5 + 0x38495ab5;
935  }
936 
937  const uint8_t *tail = (const uint8_t*)(data + nblocks*16);
938 
939  uint64_t k1 = 0;
940  uint64_t k2 = 0;
941 
942  switch(len & 15) {
943  case 15: k2 ^= ((uint64_t)tail[14]) << 48;
944  case 14: k2 ^= ((uint64_t)tail[13]) << 40;
945  case 13: k2 ^= ((uint64_t)tail[12]) << 32;
946  case 12: k2 ^= ((uint64_t)tail[11]) << 24;
947  case 11: k2 ^= ((uint64_t)tail[10]) << 16;
948  case 10: k2 ^= ((uint64_t)tail[ 9]) << 8;
949  case 9: k2 ^= ((uint64_t)tail[ 8]) << 0;
950  k2 *= c2;
951  k2 = ROTL64(k2,33);
952  k2 *= c1;
953  h2 ^= k2;
954 
955  case 8: k1 ^= ((uint64_t)tail[ 7]) << 56;
956  case 7: k1 ^= ((uint64_t)tail[ 6]) << 48;
957  case 6: k1 ^= ((uint64_t)tail[ 5]) << 40;
958  case 5: k1 ^= ((uint64_t)tail[ 4]) << 32;
959  case 4: k1 ^= ((uint64_t)tail[ 3]) << 24;
960  case 3: k1 ^= ((uint64_t)tail[ 2]) << 16;
961  case 2: k1 ^= ((uint64_t)tail[ 1]) << 8;
962  case 1: k1 ^= ((uint64_t)tail[ 0]) << 0;
963  k1 *= c1;
964  k1 = ROTL64(k1,31);
965  k1 *= c2;
966  h1 ^= k1;
967  };
968 
969  h1 ^= len; h2 ^= len;
970 
971  h1 += h2;
972  h2 += h1;
973 
974  h1 = fmix64(h1);
975  h2 = fmix64(h2);
976 
977  h1 += h2;
978  h2 += h1;
979 
980  return h1;
981 }

984  * MurmurHash3 the 64bit variant that hashes the pointer itself
985  */
986 uint64_t hashtable_hash_ptr(const void *key, int len, uint32_t seed)
987 {
988  const int nblocks = len / 4;
989 
990  uint64_t h1 = seed;
991  uint64_t h2 = seed;
992 
993  const uint64_t c1 = BIG_CONSTANT(0x87c37b91114253d5);
994  const uint64_t c2 = BIG_CONSTANT(0x4cf5ad432745937f);
995 
996  int i;
997  for (i = 0; i < nblocks; i++) {
998  uint64_t k1 = ((uintptr_t) key >> (2 * i)) & 0xff;
999  uint64_t k2 = ROTL64(k1, 13);
1000 
1001  k1 *= c1;
1002  k1 = ROTL64(k1,31);
1003  k1 *= c2;
1004  h1 ^= k1;
1005  h1 = ROTL64(h1,27);
1006  h1 += h2;
1007  h1 = h1 * 5 + 0x52dce729;
1008 
1009  k2 *= c2;
1010  k2 = ROTL64(k2,33);
1011  k2 *= c1;
1012  h2 ^= k2;
1013  h2 = ROTL64(h2,31);
1014  h2 += h1;
1015  h2 = h2 * 5 + 0x38495ab5;
1016  }
1017 
1018  /* Since the pointers are power of two length
1019  * we don't need a tail mix */
1020 
1021  h1 ^= len; h2 ^= len;
1022 
1023  h1 += h2;
1024  h2 += h1;
1025 
1026  h1 = fmix64(h1);
1027  h2 = fmix64(h2);
1028 
1029  h1 += h2;
1030  h2 += h1;
1031
