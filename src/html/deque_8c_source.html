194 bool deque_add_at(Deque *deque, void *element, size_t index)
195 {
196  if (index >= deque->size)
197  return false;
198 
199  if (deque->capacity == deque->size && !expand_capacity(deque))
200  return false;
201 
202  const size_t c = deque->capacity - 1;
203  const size_t l = deque->last & c;
204  const size_t f = deque->first & c;
205  const size_t p = (deque->first + index) & c;
206 
207  if (index == 0)
208  return deque_add_first(deque, element);
209 
210  if (index == c)
211  return deque_add_last(deque, element);
212 
213  if (index <= (deque->size / 2) - 1) {
214  if (p < f || f == 0) {
215  /* _________________________________
216  * | 1 | 2 | 3 | 4 | 5 | . | . | 6 |
217  * ---------------------------------
218  * (p) <-- L F
219  *
220  * Left circular shift from (p)
221  */
222  const size_t r_move = (f != 0) ? c - f + 1 : 0;
223  const size_t l_move = p;
224 
225  void *e_first = deque->buffer[0];
226 
227  if (f != 0) {
228  memmove(&(deque->buffer[f - 1]),
229  &(deque->buffer[f]),
230  r_move * sizeof(void*));
231  }
232  if (p != 0) {
233  memmove(&(deque->buffer[0]),
234  &(deque->buffer[1]),
235  l_move * sizeof(void*));
236  }
237  deque->buffer[c] = e_first;
238  } else {
239  memmove(&(deque->buffer[f - 1]),
240  &(deque->buffer[f]),
241  index * sizeof(void*));
242  }
243  deque->first = (deque->first - 1) & c;
244  } else {
245  if (p > l || l == c) {
246  /* _________________________________
247  * | 1 | . | . | 6 | 5 | 4 | 3 | 2 |
248  * ---------------------------------
249  * L F (p) -->
250  *
251  * Circular right shift from (p)
252  */
253  void* e_last = deque->buffer[c];
254 
255  if (p != c) {
256  memmove(&(deque->buffer[p + 1]),
257  &(deque->buffer[p]),
258  (c - p) * sizeof(void*));
259  }
260  if (l != c) {
261  memmove(&(deque->buffer[1]),
262  &(deque->buffer[0]),
263  (l + 1) * sizeof(void*));
264  }
265  deque->buffer[0] = e_last;
266  } else {
267  memmove(&(deque->buffer[p + 1]),
268  &(deque->buffer[p]),
269  (deque->size - index) * sizeof(void*));
270  }
271  deque->last = (deque->last + 1) & c;
272  }
273  deque->buffer[p] = element;
274  deque->size++;
275 
276  return true;
277 }

346 
347  void *removed = deque->buffer[index];
348 
349  if (index == 0)
350  return deque_remove_first(deque);

352  if (index == c)

354 
355  if (index <= (deque->size / 2) - 1) {
356  if (p < f) {
357  void *e = deque->buffer[c];

359  if (f != c) {

365  memmove(&(deque->buffer[1]),

372  &(deque->buffer[f]),

376  } else {

378  void *e = deque->buffer[0];

627 void deque_reverse(Deque *deque)
628 {
629  size_t i;
630  size_t j;
631  size_t s = deque->size;
632  size_t c = deque->capacity - 1;
633 
634  size_t first = deque->first;
635 
636  for (i = 0, j = s - 1; i < (s - 1) / 2; i++, j--) {
637  size_t f = (first + i) & c;
638  size_t l = (first + j) & c;
639 
640  void *tmp = deque->buffer[f];
641  deque->buffer[f] = deque->buffer[l];
642  deque->buffer[l] = tmp;
643  }
644 }

761 static void copy_buffer(Deque *deque, void **buff, void *(*cp) (void *))
762 {
763  if (cp == NULL) {
764  if (deque->last > deque->first) {
765  memcpy(buff,
766  &(deque->buffer[deque->first]),
767  deque->size * sizeof(void*));
768  } else {
769  size_t l = deque->last;
770  size_t e = deque->capacity - deque->first;
771 
772  memcpy(buff,
773  &(deque->buffer[deque->first]),
774  e * sizeof(void*));
775 
776  memcpy(&(buff[e]),
777  deque->buffer,
778  l * sizeof(void*));
779  }
780  } else {
781  size_t i;
782  for (i = 0; i < deque->size; i++) {
783  size_t p = (deque->first + i) & (deque->capacity - 1);
784  buff[i] = cp(deque->buffer[p]);

786  }
